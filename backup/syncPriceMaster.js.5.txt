const asyncHandler = require("express-async-handler");
const { releaseLock, acquireLock } = require("../../models/lock.model");
const axios = require("axios");
const Region = require("../../models/region.model");
const path = require("path");
const fs = require("fs");
const FormData = require("form-data");
const { SERVER_URL } = require("../../config/server.config");
const PriceCSV = require("../../models/priceCsv.model");
const Product = require("../../models/product.model");
const Price = require("../../models/price.model");
const PQueue = require("p-queue").default;

const syncPriceMaster = asyncHandler(async (req, res) => {
  console.log("[syncPriceMaster] ⏳ Acquiring lock...");
  if (!(await acquireLock("syncPriceMaster"))) {
    res.status(400);
    throw new Error("Another sync is in progress. Please try again later.");
  }
  console.log("[syncPriceMaster] 🔒 Lock acquired.");

  try {
    // Date handling
    const formatDate = (d) => {
      const [y, m, day] = d.toISOString().split("T")[0].split("-");
      return `${day}.${m}.${y}`;
    };

    let currentDate = formatDate(new Date());
    let previousDate = formatDate(new Date());

    if (req.query.previousDate && req.query.currentDate) {
      if (new Date(req.query.previousDate) > new Date(req.query.currentDate)) {
        res.status(400);
        throw new Error("Previous date cannot be greater than current date");
      }
      previousDate = req.query.previousDate;
      currentDate = req.query.currentDate;
    }

    // Load data and regions in parallel
    const [data, regions] = await Promise.all([
      // loadDataFromFile(),
      loadDataFromApi({
        currentDate,
        previousDate,
      }),
      Region.find({}).populate("stateId", "").lean(),
    ]);

    if (!regions.length) {
      res.status(404);
      throw new Error("No regions found");
    }

    if (!data.length) {
      res.status(404);
      throw new Error("No data found for the given date range");
    }

    console.log(`[syncPriceMaster] 📦 Loaded ${data.length} items`);

    // Build lookup maps
    const { regionCodeToId, stateCodeToRegion, vskuToInfo } = buildLookupMaps(
      regions,
      data
    );

    // Deduplicate and get product IDs in parallel
    const uniqueData = deduplicateByVSKU(data);
    const productCodeToId = await getProductIds(
      uniqueData.map((item) => item.VSKU)
    );

    console.log(`[syncPriceMaster] ✂️ Processing ${uniqueData.length} items`);

    // Process prices with optimized batching
    const priceData = await processPricesOptimized(
      uniqueData,
      stateCodeToRegion
    );

    // Build final prices array
    const finalPrices = await buildFinalPricesOptimized(
      priceData,
      vskuToInfo,
      productCodeToId,
      regionCodeToId
    );

    if (!finalPrices.length) {
      res.status(404);
      throw new Error("No valid prices found after processing");
    }

    console.log(`[syncPriceMaster] ✅ Generated ${finalPrices.length} prices`);

    // Generate and upload CSV
    const csvContent = generateCSV(finalPrices);
    const uploadResult = await uploadCSV(csvContent, currentDate);

    // Save to DB and trigger bulk update in parallel
    const [priceCSV] = await Promise.all([
      PriceCSV.create({
        url: { cronURL: uploadResult.secure_url, modifiedURL: null },
        status: "Pending",
      }),
      axios.put(`${SERVER_URL}/api/v1/price/bulk-update-status`, {}),
    ]);

    console.log("[syncPriceMaster] 🎉 Complete!");
    res.status(200).json({
      message: "Price master data synced and uploaded successfully",
      data: { urlData: uploadResult, priceCSV },
    });
  } catch (err) {
    console.error("[syncPriceMaster] ❌ Error:", err.message);
    res.status(res.statusCode === 200 ? 500 : res.statusCode);
    throw err;
  } finally {
    await releaseLock("syncPriceMaster");
    console.log("[syncPriceMaster] 🔓 Lock released.");
  }
});

// Optimized helper functions
function loadDataFromFile() {
  try {
    const filePath = path.join(
      __dirname,
      "../../script/03_Scripts/10_Donwload_Product_Master/reports/unique_products_2025-07-17T11-26-32-889Z.json"
    );
    return JSON.parse(fs.readFileSync(filePath, "utf8"));
  } catch (err) {
    throw new Error(`Failed to read file: ${err.message}`);
  }
}

// Load data from API
async function loadDataFromApi({ currentDate, previousDate }) {
  const masterUrl =
    `http://182.75.250.216:8000/sap/opu/odata/` +
    `sap/ZRUPA_PSSPL_MAT_API1_SRV/HeaderSet(` +
    `StartDate='${previousDate}',ToDate='${currentDate}')/` +
    `HeaderItem?$format=json`;
  const masterResp = await axios.get(masterUrl, {
    headers: { Cookie: "sap-usercontext=sap-client=100" },
  });
  let data = masterResp.data?.d?.results || [];
  if (!data.length) {
    throw new Error("No data found for the given date range");
  }
  console.log(`[loadDataFromApi] 📦 Loaded ${data.length} items from API`);
  return data;
}

function buildLookupMaps(regions, data) {
  const regionCodeToId = new Map();
  const stateCodeToRegion = new Map();
  const vskuToInfo = new Map();

  // Build region maps
  regions.forEach((r) => {
    regionCodeToId.set(r.code, r._id);
    stateCodeToRegion.set(r.stateId.code.padStart(2, "0"), {
      code: r.code,
      name: r.name,
    });
  });

  // Build product info map
  data.forEach((item) => {
    vskuToInfo.set(item.VSKU, {
      name: item.ProductName || "",
      brand: item.Info3 || "",
      category: item.Category || "",
    });
  });

  return { regionCodeToId, stateCodeToRegion, vskuToInfo };
}

function deduplicateByVSKU(data) {
  const seen = new Set();
  return data.filter((item) => {
    if (seen.has(item.VSKU)) return false;
    seen.add(item.VSKU);
    return true;
  });
}

async function getProductIds(productCodes) {
  const products = await Product.find({
    product_code: { $in: productCodes },
  })
    .select("_id product_code")
    .lean();

  const map = new Map();
  products.forEach((p) => map.set(p.product_code, p._id));
  return map;
}

// Ultra-optimized price processing
async function processPricesOptimized(data, stateCodeToRegion) {
  const SAP_BASE_URL =
    "http://182.75.250.216:8000/sap/opu/odata/sap/ZRUPA_MASSIST_MAT_REG_PR1_SRV";
  const queue = new PQueue({ concurrency: 15, interval: 100, intervalCap: 15 });

  const buildUrl = (material) =>
    `${SAP_BASE_URL}/HeaderSet(RegionCode='',Material='${material}',FromDate='01.01.2019',ToDate='31.12.9999')/HeaderItem?$format=json`;

  const fetchRegionPrices = async (material) => {
    try {
      const { data } = await axios.get(buildUrl(material), {
        headers: { Cookie: "sap-usercontext=sap-client=100" },
        timeout: 8000,
      });
      return data?.d?.results ?? [];
    } catch {
      return [];
    }
  };

  const processItem = async (item) => {
    const pcsInBox = getPcsInBox(item);
    const regionPrices = [];

    try {
      const rows = await queue.add(() => fetchRegionPrices(item.VSKU));

      const regionMap = new Map();
      rows.forEach((row) => {
        const price = Math.max(0, Number((row.Kbetr || 0) / pcsInBox));
        const region = stateCodeToRegion.get(row.Regio) || {
          code: "",
          name: "",
        };
        const key = `${row.Regio}-${region.code}`;

        if (!regionMap.has(key) || price > regionMap.get(key).price) {
          regionMap.set(key, {
            region_code_external: row.Regio,
            region_name_external: row.Bezei,
            region_code_internal: region.code,
            region_name_internal: region.name,
            price: Number(price.toFixed(2)),
          });
        }
      });

      regionPrices.push(...regionMap.values());
    } catch (err) {
      console.warn(`Failed to fetch prices for ${item.VSKU}:`, err.message);
    }

    return {
      product_code: item.VSKU,
      MRP: getMRP(item),
      DLP: getDLP(item),
      regionPrices,
    };
  };

  // Process in optimized batches
  const BATCH_SIZE = 500;
  const results = [];

  for (let i = 0; i < data.length; i += BATCH_SIZE) {
    const batch = data.slice(i, i + BATCH_SIZE);
    console.log(
      `[processPrices] 📦 Batch ${Math.floor(i / BATCH_SIZE) + 1}/${Math.ceil(
        data.length / BATCH_SIZE
      )}`
    );

    const batchResults = await Promise.allSettled(batch.map(processItem));
    results.push(
      ...batchResults
        .filter((r) => r.status === "fulfilled")
        .map((r) => r.value)
    );
  }

  return results;
}

async function buildFinalPricesOptimized(
  priceData,
  vskuToInfo,
  productCodeToId,
  regionCodeToId
) {
  const tomorrow = new Date();
  tomorrow.setDate(tomorrow.getDate() + 1);
  const effectiveDate = tomorrow
    .toLocaleDateString("en-IN", {
      timeZone: "Asia/Kolkata",
      day: "2-digit",
      month: "2-digit",
      year: "numeric",
    })
    .split("/")
    .join("-");

  // Batch RLP lookups
  const nationalQueries = [];
  const regionalQueries = [];

  priceData.forEach((item) => {
    const product_id = productCodeToId.get(item.product_code);
    if (product_id) {
      nationalQueries.push({ productId: product_id, price_type: "national" });
      item.regionPrices.forEach((region) => {
        const region_id = regionCodeToId.get(region.region_code_internal);
        if (region_id) {
          regionalQueries.push({
            productId: product_id,
            regionId: region_id,
            price_type: "regional",
          });
        }
      });
    }
  });

  // Fetch all RLP data in parallel
  const [nationalPrices, regionalPrices] = await Promise.all([
    Price.find({
      $or: nationalQueries,
      status: true,
    })
      .select("productId rlp_price")
      .lean(),
    Price.find({
      $or: regionalQueries,
      status: true,
    })
      .select("productId regionId rlp_price")
      .lean(),
  ]);

  // Build RLP lookup maps
  const nationalRLPMap = new Map();
  const regionalRLPMap = new Map();

  nationalPrices.forEach((p) => {
    nationalRLPMap.set(p.productId.toString(), p.rlp_price);
  });

  regionalPrices.forEach((p) => {
    regionalRLPMap.set(
      `${p.productId.toString()}-${p.regionId.toString()}`,
      p.rlp_price
    );
  });

  // Build final prices array
  const finalPrices = [];

  priceData.forEach((item) => {
    const productInfo = vskuToInfo.get(item.product_code) || {};
    const product_id = productCodeToId.get(item.product_code);

    if (!product_id || !item.DLP || parseFloat(item.DLP) <= 0) return;

    const basePrice = {
      product_id,
      "Product Code": item.product_code,
      "Product Name": productInfo.name,
      "Product Brand": productInfo.brand,
      "Product Category": productInfo.category,
      MRP: item.MRP,
      "Effective Date": effectiveDate,
      "Distributor Code": "",
      "Distributor Name": "",
    };

    // National price
    const nationalRLP = nationalRLPMap.get(product_id.toString()) || item.DLP;
    finalPrices.push({
      ...basePrice,
      "Price Type": "national",
      "Region Code": "",
      region_id: "",
      "Region Name": "",
      DLP: item.DLP,
      RLP: nationalRLP,
      "RLP Type": nationalRLP === item.DLP ? "new" : "old",
    });

    // Regional prices
    item.regionPrices.forEach((region) => {
      const region_id = regionCodeToId.get(region.region_code_internal);
      if (!region_id || region.price <= 0) return;

      const regionalRLP =
        regionalRLPMap.get(
          `${product_id.toString()}-${region_id.toString()}`
        ) || region.price;

      finalPrices.push({
        ...basePrice,
        "Price Type": "regional",
        "Region Code": region.region_code_internal,
        region_id,
        "Region Name": region.region_name_internal,
        DLP: region.price,
        RLP: regionalRLP,
        "RLP Type": regionalRLP === region.price ? "new" : "old",
      });
    });
  });

  return finalPrices;
}

function generateCSV(finalPrices) {
  const csvHeader = [
    "Price Type",
    "Region Code",
    "Region Name",
    "Distributor Code",
    "Distributor Name",
    "Product Code",
    "Product Name",
    "Product Brand",
    "Product Category",
    "MRP",
    "DLP",
    "RLP",
    "Effective Date",
    "RLP Type",
  ];

  const escapeCsvValue = (value) => {
    if (value == null) return "";
    const str = String(value);
    return /[",\n\r]/.test(str) ? `"${str.replace(/"/g, '""')}"` : str;
  };

  const csvRows = [
    csvHeader.join(","),
    ...finalPrices.map((row) =>
      csvHeader.map((field) => escapeCsvValue(row[field])).join(",")
    ),
  ];

  return csvRows.join("\n");
}

async function uploadCSV(csvContent, currentDate) {
  const uploadsDir = path.join(__dirname, "../../uploads");
  if (!fs.existsSync(uploadsDir)) {
    fs.mkdirSync(uploadsDir, { recursive: true });
  }

  const csvFileName = `price_master_${currentDate}.csv`;
  const csvFilePath = path.join(uploadsDir, csvFileName);

  fs.writeFileSync(csvFilePath, csvContent);

  try {
    const formData = new FormData();
    formData.append("my_file", fs.createReadStream(csvFilePath));
    formData.append("fileName", csvFileName);

    const result = await axios.post(
      `${SERVER_URL}/api/v1/cloudinary/upload`,
      formData,
      { headers: formData.getHeaders() }
    );

    return result.data;
  } finally {
    try {
      fs.unlinkSync(csvFilePath);
    } catch (err) {
      console.warn("Failed to delete temp file:", err.message);
    }
  }
}

// Utility functions
const getPcsInBox = (item) => {
  const raw = String(item.ProductExntension1 || "").trim();
  const num = parseFloat(raw.replace(/[^0-9.]/g, "")) || 1;
  return num > 0 ? num : 1;
};

const getMRP = (item) => {
  const MRP = Number(parseFloat(item.SalePrice || "0.00")).toFixed(2);
  return isNaN(MRP) || MRP < 0 ? "0.00" : MRP;
};

const getDLP = (item) => {
  const UnitPrice = Number(parseFloat(item.UnitPrice || "0.00")).toFixed(2);
  const pcsInBox = getPcsInBox(item);
  const DLP = Number(UnitPrice / pcsInBox).toFixed(2);
  return isNaN(DLP) || DLP < 0 ? "0.00" : DLP;
};

module.exports = { syncPriceMaster };
