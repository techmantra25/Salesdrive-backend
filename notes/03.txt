const mapBeatIdToEmployeeId = asyncHandler(async (req, res) => {
  try {
    const { beatIds } = req.body;
    const { id } = req.params;

    const employee = await Employee.findById(id);
    if (!employee) {
      res.status(404);
      throw new Error("Employee not found");
    }

    const existingBeatIds = employee.beatId.map((beatId) => beatId.toString());

    console.log("existingBeatIds", existingBeatIds);

    const removedBeatIds = existingBeatIds.filter(
      (beatId) => !beatIds.includes(beatId)
    );

    console.log("removedBeatIds", removedBeatIds);

    const newBeatIds = beatIds.filter(
      (beatId) => !existingBeatIds.includes(beatId)
    );

    // Check if any new normal beats are already occupied
    const occupiedNormalBeats = await Beat.find({
      _id: { $in: newBeatIds },
      isOccupied: true,
      beat_type: "normal",
    });

    if (occupiedNormalBeats.length > 0) {
      res.status(400);
      throw new Error("One or more normal beats are already occupied");
    }

    // Update employee's beatId array
    employee.beatId = [...new Set([...existingBeatIds, ...beatIds])];
    const updatedEmployee = await employee.save();

    // Update beats: add new employee to new beats
    await Beat.updateMany(
      { _id: { $in: newBeatIds } },
      { $set: { isOccupied: true }, $addToSet: { employeeId: employee._id } }
    );

    // Update beats: remove employee from removed beats
    await Beat.updateMany(
      { _id: { $in: removedBeatIds }, employeeId: employee._id },
      { $pull: { employeeId: employee._id } }
    );

    // Set isOccupied to false for beats with no employees
    await Beat.updateMany(
      { _id: { $in: removedBeatIds }, employeeId: { $size: 0 } },
      { $set: { isOccupied: false } }
    );

    return res.status(200).json({
      status: 200,
      message: "Employee updated successfully",
      data: updatedEmployee,
    });
  } catch (error) {
    res.status(400).json({ error: true, message: error.message });
  }
});
